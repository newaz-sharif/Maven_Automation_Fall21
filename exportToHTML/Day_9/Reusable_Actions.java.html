12   public class Reusable_Actions {
13       //create a reusable method for WebDriver and chromeoptions
14       public static WebDriver setDriver(){
15           //setup chrome driver
16           WebDriverManager.chromedriver().setup();
17           //declare and define the chrome options
18           ChromeOptions options = new ChromeOptions();
19           //declare all the arguments you need
20           //to maximize my driver
21           options.addArguments("start-maximized");
22           //options.addArguments("fullscreen"); //for mac users
23           //set as incognito mode
24           options.addArguments("incognito");
25           //set as headless(running the driver on background)
26           //options.addArguments("headless");
27           //define the web driver you are using
28           WebDriver driver = new ChromeDriver(options);
29           return  driver;
30       }//end of webdriver method
31   
32   
33       //creating void click method for any web element
34       public static void clickMethod(WebDriver driver, String xpath, String elementName){
35           //declare local explicit wait
36           WebDriverWait wait = new WebDriverWait(driver,15);
37           System.out.println("Clicking on element " + elementName);
38           try {
39               wait.until(ExpectedConditions.visibilityOfElementLocated(By.xpath(xpath))).click();
40           } catch (Exception e) {
41               System.out.println("Unable to click on " + elementName + " Error:" + e);
42           }
43       }//end of click method
44   
45       //creating void sendkeys method for any web element
46       public static void sendKeysMethod(WebDriver driver, String xpath, String userData, String elementName){
47           //declare local explicit wait
48           WebDriverWait wait = new WebDriverWait(driver,15);
49           System.out.println("Typing on element " + elementName);
50           try {
51               WebElement element = wait.until(ExpectedConditions.visibilityOfElementLocated(By.xpath(xpath)));
52               //clear first
53               element.clear();
54               //enter user data
55               element.sendKeys(userData);
56           } catch (Exception e) {
57               System.out.println("Unable to type on " + elementName + " Error:" + e);
58           }
59       }//end of click sendKeysMethod
60   
61       //creating return getText method for any text web element
62       public static String getTextMethod(WebDriver driver, String xpath, String elementName){
63           //declare local explicit wait
64           WebDriverWait wait = new WebDriverWait(driver,15);
65           System.out.println("Capturing text on element " + elementName);
66           //declare a global variable to capture the text so I can return it
67           String result = null;
68           try {
69               WebElement element = wait.until(ExpectedConditions.visibilityOfElementLocated(By.xpath(xpath)));
70               //capture text
71               result = element.getText();
72           } catch (Exception e) {
73               System.out.println("Unable to capture text on " + elementName + " Error:" + e);
74           }
75           return result;
76       }//end of click getTextMethod
